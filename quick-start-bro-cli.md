# Quick start

# Normal run (uses cache if available)

/brow-cli

# bruno env files generated
looks for a folder with *API* 
Asset APIs/environments/cli-generated.bru

# Bruno desktop app 
open `Asset APIs ` in bruno as collection and select cli-generated as environment.

# Force fresh login
/brow-cli --force-refresh

# Which browser
/brow-cli --browser full (google chrome)
/brow-cli --browser light (google chromium) - Default

# With other options
/brow-cli --mask --verbose
/brow-cli --output json --force-refresh
